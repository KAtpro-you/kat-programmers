Variables and Data Types

Working with different data types in C++

#include <iostream>
using namespace std;

int main() {
    // Integer types
    int age = 25;
    long population = 7800000000L;
    
    // Floating point types
    double price = 99.99;
    float temperature = 23.5f;
    
    // Character and string
    char grade = 'A';
    string name = "John Doe";
    
    // Boolean
    bool isStudent = true;
    
    // Output
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    cout << "Grade: " << grade << endl;
    
    return 0;
}

Classes and Objects

Object-oriented programming with classes

#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    string name;
    int age;
    double gpa;

public:
    // Constructor
    Student(string n, int a, double g) {
        name = n;
        age = a;
        gpa = g;
    }
    
    // Methods
    void displayInfo() {
        cout << "Name: " << name << endl;
        cout << "Age: " << age << endl;
        cout << "GPA: " << gpa << endl;
    }
    
    void setGPA(double newGPA) {
        gpa = newGPA;
    }
    
    double getGPA() {
        return gpa;
    }
};

int main() {
    Student student1("Alice", 20, 3.8);
    student1.displayInfo();
    return 0;
}

STL Vectors and Algorithms

Using the Standard Template Library

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    // Create vector
    vector<int> numbers = {5, 2, 8, 1, 9, 3};
    
    // Add elements
    numbers.push_back(7);
    numbers.push_back(4);
    
    // Sort the vector
    sort(numbers.begin(), numbers.end());
    
    // Print sorted vector
    cout << "Sorted numbers: ";
    for(int num : numbers) {
        cout << num << " ";
    }
    cout << endl;
    
    // Find element
    auto it = find(numbers.begin(), numbers.end(), 5);
    if(it != numbers.end()) {
        cout << "Found 5 at position: " << distance(numbers.begin(), it) << endl;
    }
    
    return 0;
}
