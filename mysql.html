Creating Database and Tables

Setting up database structure with tables

-- Create database
CREATE DATABASE school_management;
USE school_management;

-- Create students table
CREATE TABLE students (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE,
    age INT,
    enrollment_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create courses table
CREATE TABLE courses (
    id INT AUTO_INCREMENT PRIMARY KEY,
    course_name VARCHAR(100) NOT NULL,
    course_code VARCHAR(10) UNIQUE,
    credits INT DEFAULT 3,
    instructor VARCHAR(100)
);

CRUD Operations

Basic database operations - Create, Read, Update, Delete

-- INSERT data
INSERT INTO students (first_name, last_name, email, age, enrollment_date)

JOINs and Relationships

Connecting tables with JOINs and foreign keys

-- Create enrollment table for many-to-many relationship
CREATE TABLE enrollments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id INT,
    course_id INT,
    grade CHAR(2),
    enrollment_date DATE,
    FOREIGN KEY (student_id) REFERENCES students(id),
    FOREIGN KEY (course_id) REFERENCES courses(id)
);

-- INNER JOIN - Students with their enrollments
SELECT 
    s.first_name,
    s.last_name,
    c.course_name,
    e.grade
FROM students s
INNER JOIN enrollments e ON s.id = e.student_id
INNER JOIN courses c ON e.course_id = c.id;

-- LEFT JOIN - All students, including those not enrolled
SELECT 
    s.first_name,
    s.last_name,
    c.course_name
FROM students s
LEFT JOIN enrollments e ON s.id = e.student_id
LEFT JOIN courses c ON e.course_id = c.id;

Aggregate Functions and Grouping

Advanced queries with calculations and data analysis

-- Aggregate functions
SELECT COUNT(*) as total_students FROM students;
SELECT AVG(age) as average_age FROM students;
SELECT MIN(age) as youngest, MAX(age) as oldest FROM students;

-- GROUP BY with aggregate functions
SELECT 
    c.course_name,
    COUNT(e.student_id) as enrolled_students,
    AVG(CASE 
        WHEN e.grade = 'A' THEN 4
        WHEN e.grade = 'B' THEN 3
        WHEN e.grade = 'C' THEN 2
        WHEN e.grade = 'D' THEN 1
        ELSE 0 
    END) as average_gpa
FROM courses c
LEFT JOIN enrollments e ON c.id = e.course_id
GROUP BY c.id, c.course_name
HAVING enrolled_students > 0;

-- Subquery example
SELECT first_name, last_name, age
FROM students
WHERE age > (SELECT AVG(age) FROM students);
